  • Factorielle

    Une fonction qui calcule la factorielle d'un nombre donné.


    Résultat:

    Code de la solution : 
    function calculateFactorial() {
    var num = parseInt(document.getElementById('factorial-num').value);
    if (isNaN(num) || num < 0) {
        document.getElementById('factorial-result').innerText = "Veuillez entrer un nombre positif.";
        return;
    }
    var result = 1;
    for (var i = 1; i <= num; i++) {
        result *= i;
    }
    document.getElementById('factorial-result').innerText = result;
}

  • Inverser une chaîne

    Une fonction qui inverse une chaîne de caractères donnée.


    Résultat:

    Code de la solution : 
      function reverseString() {
    var str = document.getElementById('reverse-string').value;
    var reversed = "";
    for (var i = str.length - 1; i >= 0; i--) {
        reversed += str[i];
    }
    document.getElementById('reversed-result').innerText = reversed;
}

  • Compter les Voyelles

    Une fonction qui compte le nombre de voyelles dans une chaîne de caractères donnée.


    Résultat:

    Code de la solution : 
    function countVowels() {
    var str = document.getElementById('vowel-string').value.toLowerCase();
    var vowels = "aeiou";
    var count = 0;
    for (var i = 0; i < str.length; i++) {
        if (vowels.indexOf(str[i]) !== -1) {
            count++;
        }
    }
    document.getElementById('vowel-count-result').innerText = count;
}

  • Moyenne

    Une fonction qui calcule la moyenne d'une liste de nombres séparés par des virgules.


    Résultat:

    Code de la solution : 
    function calculateAverage() {
    var input = document.getElementById('average-numbers').value;
    var numbers = input.split(',');
    var sum = 0;
    for (var i = 0; i < numbers.length; i++) {
        var num = parseFloat(numbers[i]);
        if (isNaN(num)) {
            document.getElementById('average-result').innerText = "Entrez uniquement des nombres séparés par des virgules.";
            return;
        }
        sum += num;
    }
    var average = sum / numbers.length;
    document.getElementById('average-result').innerText = average.toFixed(2);
}

  • Nombre Premier

    Une fonction qui vérifie si un nombre est premier.


    Résultat:

    Code de la solution : 
            function isPrime() {
            const num = parseInt(document.getElementById('prime-number').value);
            if (num <= 1) {
                document.getElementById('prime-result').innerText = "Non, ce n'est pas un nombre premier.";
                return;
            }
            for (let i = 2; i <= Math.sqrt(num); i++) {
                if (num % i === 0) {
                    document.getElementById('prime-result').innerText = "Non, ce n'est pas un nombre premier.";
                    return;
                }
            }
            document.getElementById('prime-result').innerText = "Oui, c'est un nombre premier.";
        }

            const num = parseInt(document.getElementById('multiplication-number').value);
